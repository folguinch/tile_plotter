181    def get_normalization(self, vmin: Optional[float] = None,
182                          vmax: Optional[float] = None,
183                          vcenter: Optional[float] = None) -> Colormap:
184        """Determine the normalization of the color stretch.
185
186        Args:
187          vmin: optional; scale minimum.
188          vmax: optional; scale maximum.
189          vcenter: optional; `midnorm` scale center.
190        """
191        if vmin is None:
192            vmin = self.vmin
193        if vmax is None:
194            vmax = self.vmax
195        if vcenter is None:
196            vcenter = self.vcenter
197        if self.stretch == 'log':
198            return ImageNormalize(vmin=vmin, vmax=vmax,
199                                  stretch=LogStretch(a=self.a))
200        elif self.stretch == 'midnorm':
201            return TwoSlopeNorm(vcenter, vmin=vmin, vmax=vmax)
202        else:
203            return ImageNormalize(vmin=vmin, vmax=vmax, stretch=LinearStretch())

246    def get_vlim(self, axis: int = 1) -> Tuple[float, float]:
247        """Get limits for the intensity axis.

249        Args:

293        """Check units of the values defining the intensity scale."""

299            if self.vcenter is not None:

304    def generate_cbar2(self, unit_fmt: str = '({:latex_inline})'):
305        """Fill the properties of second colorbar."""

308        self.vmax_cbar2 = self.vmax.to(self.unit_cbar2,

310        self.ticks_cbar2 = self.ticks.to(self.unit_cbar2,
