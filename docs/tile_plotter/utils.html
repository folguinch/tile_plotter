 28def to_sigfig(x: Quantity, sigfig: int = 2) -> Quantity:
 29    """Round value to sigfig significant figures."""
 30    try:
 31        val = x.value
 32    except AttributeError:
 33        val = x
 34    ndigits = -int(np.floor(np.log10(val))) + (sigfig - 1)
 35    return np.round(x, ndigits)

 37def get_data_type(data: Quantity, default: Optional[str] = None) -> str:
 38    """Determine the data type based on `data` unit."""

 40        data_type = 'velocity'

 48def get_extent(data: Map,
 49               wcs: Optional['astropy.wcs.WCS'] = None) -> List[u.Quantity]:
 50    """Determine the data extent."""
 51    if wcs is not None and not hasattr(data, 'header'):
 52        image = fits.PrimaryHDU(data.value, header=wcs.to_header())
 53    elif wcs is None and not hasattr(data, 'header'):
 54        raise ValueError('Extent cannot be determined')
 55    else:
 56        image = data
 57    xaxis, yaxis = images.get_coord_axes(image)
 58
 59    return [xaxis[0], xaxis[-1], yaxis[0], yaxis[-1]]

 61def get_colorbar_ticks(vmin: u.Quantity,
 62                       vmax: u.Quantity,
 63                       a: float = 1000.,
 64                       n: int = 5,
 65                       stretch: str = 'linear',
 66                       ndigits: Optional[int] = None,
 67                       sigfig: Optional[int] = None) -> Quantity:
 68    """Calculate the tick values for the colorbar given the intensity stretch.
 69
 70    Args:
 71      vmin: the minimum of the stretch.
 72      vmax: the maximum of the stretch.
 73      a: optional; scaling for logarithmic stretch.
 74      n: optional; number of ticks.
 75      stretch: optional; stretch type.
 76      ndigits: optional; value of digits for rounding.
 77      sigfig: optional; number of significant figures.
 78
 79    Returns:
 80      A numpy array quantity with the ticks.
 81    """
 82    if stretch == 'log':
 83        def log_dist(y, a):
 84            return (10.**(y * np.log10(a + 1.)) - 1.) / a
 85        y = np.linspace(0., 1., n)
 86        x = log_dist(y, a)
 87        ticks = x*(vmax - vmin) + vmin
 88    elif stretch == 'symlog':
 89        n = n//2 + 1
 90        linthresh = min(abs(vmin.value), abs(vmax.value))
 91        linthresh = linthresh / 10.
 92        if abs(vmin) < vmax:
 93            npos = n + 1
 94            nneg = n
 95        elif abs(vmin) > vmax:
 96            npos = n
 97            nneg = n + 1
 98        else:
 99            npos = nneg = n
100        ypos = np.logspace(np.log10(linthresh), np.log10(vmax.value), npos)
101        yneg = np.logspace(np.log10(linthresh), np.log10(abs(vmin.value)), nneg,
102                           endpoint=False)
103        ticks = np.append(-yneg, [0])
104        ticks = np.unique(np.append(ticks, ypos)) * vmin.unit
105    elif stretch in ['linear', 'midnorm']:
106        ticks = np.linspace(vmin, vmax, n)
107    else:
108        raise ValueError(f'Stretch {stretch} not recognized')
109
110    # Round if needed
111    if ndigits:
112        ticks = np.unique(np.round(ticks, ndigits))
113    if sigfig:
114        ticks = np.unique(to_sigfig(ticks, sigfig=ndigits))
115
116    return ticks

118def auto_vmin(data: u.Quantity,
119              rms: Optional[u.Quantity] = None,
120              nrms: float = 0.8,
121              velocity_fraction: Optional[float] = None,
122              data_type: Optional[str] = None,
123              sigfig: Optional[int] = None,
124              log: Callable = print) -> u.Quantity:
125    """Determine vmin from the data.
126
127    The `data_type` is assumed to be `intensity` unless specified or if the
128    units of `data` are equivalent to `km/s`, in which case it is set to
129    `velocity`. If `data_type=intensity`, it calculates the rms from the data
130    if not given. The value of `vmin` is `rms * nrms` in this case. If
131    `data_type=velocity`, `vmin` is a fraction of the minimum value of the data.
132
133    If `velocity_fraction` is `None`, the default behaviour is to subtract 2% of
134    the minumum value.
135
136    Args:
137      data: data to calculate vmin from.
138      rms: optional; rms of the data.
139      nrms: optional; number of rms values for vmin.
140      velocity_fraction: optional; fraction of the minimum value of the data to
141        use if `data_type=velocity`.
142      data_type: optional; type of map.
143      sigfig: optional; number of significant figures of output.
144      log: optional; logging function.
145    """
146    # Verify data type
147    data_type = get_data_type(data, default=data_type)
148
149    # Get vmin
150    if data_type.lower() in ['intensity', 'pvmap']:
151        if rms is None:
152            rms = apystats.mad_std(data, ignore_nan=True)
153            log(f'MAD std = {rms:.3e}')
154        else:
155            log(f'Input rms = {rms:.3e}')
156        vmin = nrms * rms
157    elif data_type.lower() == 'velocity':
158        vmin = np.nanmin(data)
159        if velocity_fraction is None:
160            vmin = vmin - np.abs(0.02 * vmin)
161        else:
162            vmin = vmin * velocity_fraction
163    else:
164        raise KeyError(f'Map type {data_type} not recognized')
165
166    if sigfig is not None:
167        return to_sigfig(vmin.to(data.unit), sigfig=sigfig)
168    else:
169        return vmin.to(data.unit)

171def auto_vmax(data: u.Quantity,
172              maximum: Optional[u.Quantity] = None,
173              fraction: Optional[float] = None,
174              sigfig: Optional[int] = None) -> u.Quantity:
175    """Determine vmax from the data maximum.
176
177    If maximum is given, then this value is used instead of the data. At the
178    moment there isn't any difference based on the map type.
179
180    The default behaviour if `fraction` is `None` is to add 2% of the maximum
181    value.
182
183    Args:
184      data: data to calculate vmax from.
185      maximum: optional; maximum value.
186      fraction: optional; fraction of the maximum for vmax.
187      sigfig: optional; number of significant figures of output.
188    """
189    if maximum is not None:
190        vmax = maximum
191    else:
192        vmax = np.nanmax(data)
193
194    if fraction is None:
195        vmax = vmax + np.abs(vmax * 0.02)
196    else:
197        vmax = vmax * fraction
198
199    if sigfig is not None:
200        return to_sigfig(vmax.to(data.unit), sigfig=sigfig)
201    else:
202        return vmax.to(data.unit)

211def get_nlevels(max_val: u.Quantity, min_val: u.Quantity, stretch: str,
212                base: Optional[float] = 2.) -> int:
213    """Determine the number of levels based on the stretch.
214
215    Levels are assumed to be a multiple of min_val.
216
217    Args:
218      max_val: maximum value.
219      min_val: minimum value.
220      stretch: level stretch.
221      base: optional; base of the log levels.
222    """
223    if stretch == 'log':
224        ratio = max_val.value / min_val.to(max_val.unit).value
225        nlevels = int(np.floor(np.log(ratio) / np.log(base))) + 1
226    elif stretch == 'linear':
227        nlevels = int(max_val.value / min_val.to(max_val.unit).value)
228    else:
229        raise NotImplementedError
230
231    return nlevels

233def auto_levels(data: Optional[u.Quantity] = None,
234                rms: Optional[u.Quantity] = None,
235                max_val: Optional[u.Quantity] = None,
236                min_val: Optional[u.Quantity] = None,
237                nsigma: float = 5.,
238                stretch: str = 'linear',
239                base: float = 2.,
240                nlevels: Optional[int] = None,
241                min_nlevels: int = 1,
242                negative_nsigma: Optional[float] = None,
243                min_base: float = 1.1,
244                nsigmalevel: Optional = None,
245                log: Callable = print) -> u.Quantity:
246    """Determine the levels for contour plots.
247
248    Args:
249      data: optional; data to extract the levels from.
250      rms: optional; rms of the data.
251      max_val: optional; maximum contour.
252      min_val: optional; minumum contour.
253      nsigma: optional; minimum contour over rms value.
254      stretch: optional; stretch of the contours.
255      negative_nsigma: optional; maximum contour for negative levels.
256      base: optional; logarithmic base.
257      nlevels: optional; number of levels.
258      min_nlevels: optional; minimum number of levels.
259      min_base: optional; minimum base for refining.
260      nsigmalevel: optional; only one level at this nsigma value.
261      log: optional; logging function.
262    """
263    # Value checks
264    log('Calculating levels')
265    if stretch == 'linear':
266        base = 1
267    elif min_base > base:
268        log('Minimum base > base, setting min_base = base')
269        min_base = base
270    if nlevels and nlevels < min_nlevels:
271        log('Minimum nlevels > nlevels, setting nlevels = min_nlevels')
272        nlevels = min_nlevels
273    if data is not None and max_val is None:
274        max_val = np.nanmax(data)
275
276    # Determine rms
277    log('Determining levels from data:')
278    if rms is None and data is not None:
279        log('Using automatic rms for levels')
280        rms = apystats.mad_std(data, ignore_nan=True)
281    elif rms is None and data is None:
282        raise ValueError('Could not determine levels')
283    rms = to_sigfig(rms)
284    log(f'Getting levels from rms = {rms}')
285
286    # One level at nsigmalevel
287    if nlevels and nlevels == 1:
288        log(f'Setting only one level at {nsigmalevel} sigma')
289        levels = np.array([nsigma * rms.value]) * rms.unit
290        log(f'Levels = {levels}')
291        return levels
292
293    # Limits
294    if max_val is not None:
295        if stretch == 'linear':
296            base_level = rms
297        else:
298            base_level = nsigma * rms
299        max_val = max_val.to(rms.unit)
300        if nlevels is None:
301            nlevels = get_nlevels(max_val, base_level, stretch, base=base)
302        if nlevels == 0:
303            log(f'No contours over {nsigma}rms')
304            log('Using one level at rms level')
305            nlevels = 1
306            nsigma = 1
307        elif nlevels < min_nlevels:
308            if (stretch == 'log' and base == min_base) or stretch == 'linear':
309                log('Minimum number of levels not achieved with min_base')
310                log(f'Setting min_nlevels value = {nlevels}')
311                min_nlevels = nlevels
312            elif stretch == 'log' and base > min_base:
313                log('Minimum number of levels not achieved, refining base')
314                ratio = max_val.value / base_level.value
315                new_base = to_sigfig(np.exp(np.log(ratio) / (min_nlevels-1)))
316                if new_base < min_base:
317                    log(f'Estimated base {new_base} smaller than min_base')
318                    log(f'Changing base to {min_base}')
319                    base = min_base
320                else:
321                    log(f'New base = {new_base}')
322                    base = new_base
323                nlevels = get_nlevels(max_val, base_level, stretch, base=base)
324        log(f'Number of levels = {nlevels}')
325    elif nlevels:
326        log(f'Using nlevels = {nlevels}')
327    else:
328        raise ValueError('Could not determine nlevels')
329
330    # Calculate positive levels
331    if stretch == 'log':
332        # Geometric progression
333        aux_levels = [nsigma * base**i for i in range(nlevels)]
334    elif stretch == 'linear':
335        aux_levels = list(range(int(nsigma), nlevels))
336
337    # Negative contours
338    if negative_nsigma is not None:
339        # pylint: disable=invalid-unary-operand-type
340        aux = -negative_nsigma * rms
341        if min_val is None and data is not None:
342            min_val = np.nanmin(data)
343            log(f'Minimum data: {min_val}')
344            log(f'Start negative level (-{negative_nsigma}sigma): {aux}')
345            loop = True
346        elif min_val is None and data is None:
347            log('Could not determine minimum value')
348            log('Skipping negative levels')
349            loop = False
350        else:
351            log(f'Minimum data: {min_val}')
352            log(f'Start negative level (-{negative_nsigma}sigma): {aux}')
353            loop = True
354
355        # Get negative levels
356        i = -1
357        while loop:
358            factor = i * abs(negative_nsigma)
359            newlevel = factor * rms
360            if newlevel <= min_val:
361                break
362            elif factor not in aux_levels:
363                aux_levels = [factor] + aux_levels
364            if stretch == 'log':
365                i = i * base
366            elif stretch == 'linear':
367                i = i - 1
368
369    # Levels
370    levels = np.array(aux_levels) * rms
371    log(f'Levels/rms = {aux_levels}')
372    log(f'Levels = {levels}')
373
374    return levels

400def get_artist_positions(values: str, artist: str,
401                         separator: str = ',',
402                         xycoords: str = 'data',
403                         phys_frame: str = 'sky') -> tuple:
404    """Extract the position values for the input `artist`.
405
406    It uses `artist` to determine the type of the output.
407
408    Args:
409      values: string where the positions are obtained.
410      artist: artist type.
411      separator: optional; separator between values.
412      xycoords: optional; type of coordinates.
413      phys_frame: optional; physical frame of the data (`sky` or `projection`)
414    """
415    quantity_artists = ('scatters', 'arcs', 'texts', 'arrows', 'scale',
416                        'axlines')
417    vals = values.split(separator)
418    positions = []
419    for val in vals:
420        if artist in quantity_artists and xycoords == 'data':
421            if phys_frame == 'sky':
422                try:
423                    ra, dec, frame = val.split()
424                except ValueError:
425                    ra, dec = val.split()
426                    frame = 'icrs'
427                positions.append(SkyCoord(ra, dec, frame=frame))
428            else:
429                # Emulate a SkyCoord
430                try:
431                    ra, raunit, dec, decunit = val.split()
432                    ra = u.Quantity(f'{ra} {raunit}')
433                    dec = u.Quantity(f'{dec} {decunit}')
434                except ValueError:
435                    val_split = val.split()
436                    if len(val_split) == 3:
437                        ra, dec, unit = val_split
438                        ra = u.Quantity(f'{ra} {unit}')
439                        dec = u.Quantity(f'{dec} {unit}')
440                    elif len(val_split) == 4:
441                        ra, raunit, dec, decunit = val_split
442                        ra = u.Quantity(f'{ra} {raunit}')
443                        dec = u.Quantity(f'{dec} {decunit}')
444                positions.append(LikeSkyCoord(ra, dec))
445        elif artist in ['hlines', 'vlines'] and xycoords == 'data':
446            positions.append(u.Quantity(val))
447        else:
448            positions.append(tuple(map(float, val.split())))
449
450    return positions

452def get_artist_properties(
453    artist: str,
454    config: 'configparseradv.configparser.ConfigParserAdv',
455    separator: str = ',',
456    float_props: Sequence[str] = ('size', 'width', 'height', 'angle', 's',
457                                  'alpha', 'length', 'linewidth'),
458    quantity_props: Sequence[str] = ('pa', 'slope'),
459    from_region: bool = False,
460) -> dict:
461    """Extract the properties of the artists from `config`.
462
463    This function creates a dictionary that contain the markers positions as
464    keys and a dictionary of the artist properties as its value. The keywords
465    are extracted from the option name in the `config`.
466
467    Example:
468    A configuration file with the following options:
469
470    ```init
471    scatter = pos1x pos1y, pos2x pos2y
472    scatter_color = r, w
473    ```
474
475    will be converted to:
476
477    ```python
478    properties = {'positions': ((pos1x, pos1y), (pos2x, pos2y)),
479                  'properties': ({'color': 'r'}, {'color': 'w'})}
480    ```
481
482    Each dictionary can be used as keyword input for the corresponding
483    `matplotlib` function.
484
485    Args:
486      artist: name of the artist.
487      config: configuration parser proxy.
488      separator: optional; separator between values of an config option.
489      float_props: optional; list of properties to be converted to float.
490      quantity_props: optional; list of properties that are `Quantity`.
491      from_region: optional; read positions from region?
492    """
493    # Position
494    if from_region:
495        positions = positions_from_region(config[artist])
496    else:
497        xycoords = config.get(f'{artist}_xycoords', fallback='data')
498        phys_frame = config.get(f'{artist}_physframe', fallback='sky')
499        positions = get_artist_positions(config[artist], artist,
500                                         separator=separator,
501                                         xycoords=xycoords,
502                                         phys_frame=phys_frame)
503    nprops = len(positions)
504
505    # Iterate over properties
506    props = ()
507    for opt in config:
508        # Filter
509        if not opt.startswith(artist) or opt == artist:
510            continue
511
512        # Iterate over positions
513        prop = opt.split('_')[-1]
514        if prop == 'physframe':
515            continue
516        for i in range(nprops):
517            if prop in float_props:
518                val = config.getvalue(opt, n=i, dtype=float, allow_global=True,
519                                      sep=separator)
520            elif prop in ['zorder']:
521                val = config.getvalue(opt, n=i, dtype=int, allow_global=True,
522                                      sep=separator)
523            elif prop in quantity_props:
524                val = config.getvalue(opt, n=i, dtype='quantity',
525                                      allow_global=True, sep=separator)
526            else:
527                val = config.getvalue(opt, n=i, allow_global=True,
528                                      sep=separator)
529                if 'unit' in prop:
530                    val = u.Unit(val)
531
532            # Special value
533            if val == 'none' and prop != 'fillstyle':
534                val = None
535
536            # Fill the properties tuple
537            try:
538                props[i].update({prop: val})
539            except IndexError:
540                props += ({prop: val},)
541
542    return {'positions': positions, 'properties': props}

262      log: optional; logging function.
263    """
264    # Value checks
265    log('Calculating levels')
266    if stretch == 'linear':
267        base = 1
268    elif min_base > base:
269        log('Minimum base > base, setting min_base = base')
270        min_base = base
271    if nlevels and nlevels < min_nlevels:
272        log('Minimum nlevels > nlevels, setting nlevels = min_nlevels')
273        nlevels = min_nlevels
274    if data is not None and max_val is None:
275        max_val = np.nanmax(data)
276
277    # Determine rms

283        raise ValueError('Could not determine levels')

285    log(f'Getting levels from rms = {rms}')
286
287    # One level at nsigmalevel
288    if nlevels and nlevels == 1:
289        log(f'Setting only one level at {nsigmalevel} sigma')
290        levels = np.array([nsigma * rms.value]) * rms.unit
291        log(f'Levels = {levels}')
292        return levels
293
294    # Limits
295    if max_val is not None:
296        if stretch == 'linear':
297            base_level = rms
298        else:

301        if nlevels is None:

303        if nlevels == 0:

305            log('Using one level at rms level')
306            nlevels = 1
307            nsigma = 1
308        elif nlevels < min_nlevels:
309            if (stretch == 'log' and base == min_base) or stretch == 'linear':
310                log('Minimum number of levels not achieved with min_base')
311                log(f'Setting min_nlevels value = {nlevels}')
312                min_nlevels = nlevels

314                log('Minimum number of levels not achieved, refining base')

317                if new_base < min_base:

319                    log(f'Changing base to {min_base}')
320                    base = min_base
321                else:
322                    log(f'New base = {new_base}')
323                    base = new_base
324                nlevels = get_nlevels(max_val, base_level, stretch, base=base)
325        log(f'Number of levels = {nlevels}')
326    elif nlevels:

328    else:

330
331    # Calculate positive levels
332    if stretch == 'log':
333        # Geometric progression
334        aux_levels = [nsigma * base**i for i in range(nlevels)]
335    elif stretch == 'linear':
336        aux_levels = list(range(int(nsigma), nlevels))

345            log(f'Start negative level (-{negative_nsigma}sigma): {aux}')
346            loop = True
347        elif min_val is None and data is None:
348            log('Could not determine minimum value')
349            log('Skipping negative levels')
350            loop = False
351        else:
352            log(f'Minimum data: {min_val}')
353            log(f'Start negative level (-{negative_nsigma}sigma): {aux}')
354            loop = True
355
356        # Get negative levels
357        i = -1
358        while loop:
359            factor = i * abs(negative_nsigma)
360            newlevel = factor * rms
361            if newlevel <= min_val:
362                break
363            elif factor not in aux_levels:
364                aux_levels = [factor] + aux_levels
365            if stretch == 'log':
366                i = i * base
367            elif stretch == 'linear':
368                i = i - 1
369
370    # Levels
371    levels = np.array(aux_levels) * rms
372    log(f'Levels/rms = {aux_levels}')
373    log(f'Levels = {levels}')
374
375    return levels
