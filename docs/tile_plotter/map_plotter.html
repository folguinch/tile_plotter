  32def filter_config_data(config: ConfigParserAdv, keys: Sequence,
  33                       ignore: Sequence = ('image', 'contour')) -> Dict:
  34    """Filter the data options in `config` present in `skeleton`.
  35
  36    Args:
  37      config: input configuration.
  38      keys: available keys.
  39      ignore: optional; keys to ignore.
  40
  41    Returns:
  42      A dictionary with the filtered values.
  43    """
  44    options = {}
  45    for key in keys:
  46        if key not in config or key in ignore:
  47            continue
  48        if key in ['center']:
  49            options[key] = config.getskycoord(key)
  50        elif key in ['radius', 'rms', 'levels']:
  51            options[key] = config.getquantity(key)
  52        elif key in ['self_contours', 'use_extent']:
  53            options[key] = config.getboolean(key)
  54        elif key in ['nsigma', 'negative_nsigma', 'nsigma_level',
  55                     'contour_linewidth']:
  56            options[key] = config.getfloat(key)
  57        else:
  58            aux = config.get(key).split()
  59            if len(aux) == 1:
  60                options[key] = aux[0]
  61            else:
  62                options[key] = aux
  63
  64    return options

 121    def from_config(cls,
 122                    config: ConfigParserAdv,
 123                    axis: Axes,
 124                    cbaxis: Axes,
 125                    radesys: Optional[str] = None,
 126                    **kwargs):
 127        """Create a new MapHandler from a config proxy.
 128
 129        Default `units` are taken from the class `skeleton`.
 130
 131        Args:
 132          config: configuration parser proxy.
 133          axis: matplotlib axis.
 134          cbaxis: matplotlib color bar axis.
 135          radesys: optional; projection system.
 136          kwargs: replace values in the config.
 137        """
 138        # Get axes properties
 139        axes_props = {}
 140        for opt, val in cls.skeleton.items('axes_props'):
 141            if 'unit' in opt:
 142                value = config.get(opt, vars=kwargs, fallback=val)
 143                try:
 144                    value = u.Unit(value)
 145                except ValueError:
 146                    value = None
 147            elif opt.startswith('set_') or opt.startswith('invert'):
 148                fallback = cls.skeleton.getboolean('axes_props', opt)
 149                value = config.getboolean(opt, vars=kwargs, fallback=fallback)
 150            elif opt in ['label_xpad', 'label_ypad']:
 151                value = config.getfloat(opt, vars=kwargs, fallback=float(val))
 152            else:
 153                value = config.get(opt, vars=kwargs, fallback=val)
 154            axes_props[opt] = value
 155
 156        # Get vscale
 157        vscale = {'stretch': cls.skeleton['vscale']['stretch']}
 158        for opt in cls.skeleton.options('vscale'):
 159            if opt not in config and opt not in kwargs:
 160                continue
 161            val = config.get(opt, vars=kwargs)
 162            if 'unit' in opt:
 163                try:
 164                    val = u.Unit(val)
 165                except ValueError:
 166                    val = None
 167            elif 'name' in opt or 'stretch' in opt:
 168                pass
 169            elif opt in ['ticks', 'ticklabels']:
 170                val = val.split()
 171                if opt == 'ticks':
 172                    val = np.array(list(map(float, val[:-1]))) * u.Unit(val[-1])
 173            elif opt in ['nticks']:
 174                val = int(val)
 175            elif opt in ['compute_ticks']:
 176                val = config.getboolean(opt, vars=kwargs)
 177            else:
 178                try:
 179                    split_val = val.split()
 180                    val = float(split_val[0]) * u.Unit(split_val[1])
 181                except ValueError:
 182                    pass
 183                except IndexError:
 184                    val = float(val)
 185            vscale[opt] = val
 186
 187        # Artists
 188        artists = {}
 189        for opt in cls.skeleton.options('artists'):
 190            if opt not in config and opt not in kwargs:
 191                continue
 192            # Scale is a special artist
 193            if opt == 'scale':
 194                props = ('length', 'size', 'distance')
 195                artists[opt] = get_artist_properties(opt, config,
 196                                                     float_props=(),
 197                                                     quantity_props=props)
 198            else:
 199                artists[opt] = get_artist_properties(opt, config,
 200                                                     from_region=opt=='regions')
 201
 202        return cls(axis, cbaxis, radesys=radesys, axes_props=axes_props,
 203                   vscale=vscale, artists=artists)

 229    def _validate_data(self, data: Map,
 230                       wcs: apy_wcs.WCS,
 231                       ignore_units: bool = False
 232                       ) -> Tuple[u.Quantity, apy_wcs.WCS]:
 233        """Validate input data.
 234
 235        Convert input data to a quantity with the same unit as `self.bunit` or
 236        determines `self.bunit` from header. If the unit of the data cannot be
 237        determined it is assumed to be in the same units as `self.bunit` or is
 238        considered dimensionless.
 239
 240        Args:
 241          data: input data.
 242          wcs: WCS of the data.
 243          ignore_units: optional; ignore data units?
 244        """
 245        if hasattr(data, 'unit'):
 246            # Check bunit
 247            if not ignore_units and self.vscale.unit is None:
 248                self.vscale.set_unit(data.unit)
 249                self._log.info('Setting bunit to data unit: %s', data.unit)
 250            elif not ignore_units:
 251                valdata = data.to(self.vscale.unit)
 252            else:
 253                valdata = data
 254        elif hasattr(data, 'header'):
 255            # Check bunit
 256            bunit = u.Unit(data.header.get('BUNIT', 1))
 257            valdata = np.squeeze(data.data) * bunit
 258            if not ignore_units and self.vscale.unit is None:
 259                self.vscale.set_unit(bunit)
 260                self._log.info('Setting bunit to header unit: %s', bunit)
 261            elif not ignore_units:
 262                self._log.info('Converting data unit: %s -> %s',
 263                               valdata.unit, self.vscale.unit)
 264                valdata = valdata.to(self.vscale.unit)
 265            else:
 266                pass
 267
 268            # Check wcs
 269            if wcs is None:
 270                wcs = apy_wcs.WCS(data.header, naxis=2)
 271
 272            # Set RADESYS
 273            if self.radesys is None:
 274                try:
 275                    self.radesys = data.header['RADESYS'].lower()
 276                    if self.radesys.upper() == 'J2000':
 277                        self.radesys = 'fk5'
 278                    self._log.info('Setting RADESYS: %s', self.radesys)
 279                except KeyError:
 280                    self._log.warning('Map does not have RADESYS')
 281                    self.radesys = ''
 282        elif not ignore_units:
 283            if self.vscale.unit is None:
 284                self._log.warning('Setting data as dimensionless')
 285                self.vscale.unit = u.Unit(1)
 286            else:
 287                self._log.warning('Assuming data in %s units', self.vscale.unit)
 288            valdata = data * self.vscale.unit
 289        else:
 290            self._log.warning('Setting data as dimensionless')
 291            valdata = data * u.Unit(1)
 292
 293        # Check wcs
 294        if wcs is None:
 295            self._log.warning('WCS is None')
 296
 297        return valdata, wcs

 332    def plot_map(self,
 333                 data: Map,
 334                 wcs: Optional[apy_wcs.WCS] = None,
 335                 extent: Optional[Tuple[u.Quantity]] = None,
 336                 use_extent: bool = False,
 337                 rms: Optional[u.Quantity] = None,
 338                 mask_bad: bool = False,
 339                 mask_color: str = 'w',
 340                 position: Optional['astropy.coordinates.SkyCoord'] = None,
 341                 radius: Optional[u.Quantity] = None,
 342                 shift_data: Optional[u.Quantity] = None,
 343                 self_contours: bool = False,
 344                 contour_levels: Optional[List[u.Quantity]] = None,
 345                 contour_colors: str = 'w',
 346                 contour_linewidths: Optional[float] = None,
 347                 contour_nsigma: float = 5.,
 348                 contour_negative_nsigma: Optional[float] = None,
 349                 contour_nsigmalevel: Optional[float] = None,
 350                 contour_stretch: Optional[str] = None,
 351                 **kwargs) -> None:
 352        """Plot image data.
 353
 354        The map can be recenter and zoomed using the `position` and `radius`
 355        parameters if the data `wcs` can be determined or is given.
 356
 357        Contour keywords are used only if `self_contours` are requested.
 358
 359        Args:
 360          data: input map.
 361          wcs: optional; WCS object of the map.
 362          extent: optional; `xy` range of the data (left, right, bottom, top).
 363          use_extent: optional; determine extent from data and ignore wcs.
 364          rms: optional; map noise level.
 365          mask_bad: optional; mask bad/null pixels?
 366          mask_color: optional; color for bad/null pixels.
 367          position: optional; center of the map.
 368          radius: optional; radius of the region shown.
 369          shift_data: optional; an additive constant to apply to the data.
 370          self_contours: optional; plot contours of the data too?
 371          contour_levels: optional; self contour levels.
 372          contour_colors: optional; self contour colors.
 373          contour_linewidths: optional; self contours line width.
 374          contour_nsigma: optional; level of the lowest contour.
 375          contour_negative_nsigma: optional; level of the highest negative
 376            contour.
 377          contour_nsigmalevel: optional; plot only one contour at this level
 378            times the rms value.
 379          contour_colors: optional; self contour stretch.
 380          **kwargs: optional; additional arguments for `pyplot.imshow`.
 381        """
 382        # Validate data: valdata is a quantity with self.bunit units
 383        valdata, valwcs = self._validate_data(data, wcs)
 384
 385        # Shift data
 386        if shift_data is not None:
 387            valdata = valdata + shift_data.to(valdata.unit)
 388
 389        # Check vscale and get normalization
 390        self._validate_vscale(valdata, rms=rms)
 391        norm = self.vscale.get_normalization()
 392
 393        # Check extent
 394        if extent is not None:
 395            extent_val = self._validate_extent(extent)
 396            self._log.info('Validated extent: %s', extent_val)
 397            valwcs = None
 398        elif extent is None and use_extent:
 399            extent = get_extent(data, wcs=wcs)
 400            self._log.info('Extent from data: %s', extent)
 401            extent_val = self._validate_extent(extent)
 402            self._log.info('Validated extent: %s', extent_val)
 403            valwcs = None
 404        else:
 405            extent_val = None
 406
 407        # Check wcs and re-center the image
 408        if valwcs is not None and radius is not None and position is not None:
 409            self.recenter(radius, position, valwcs)
 410        elif extent_val is not None:
 411            self.ax.set_xlim(*extent_val[:2])
 412            self.ax.set_ylim(*extent_val[2:])
 413
 414        # Plot data
 415        zorder = kwargs.setdefault('zorder', 1)
 416        if mask_bad:
 417            cmap = cm.get_cmap()
 418            cmap.set_bad(mask_color, 1.0)
 419            maskdata = np.ma.array(valdata.value, mask=np.isnan(data))
 420            self.im = self.ax.imshow(maskdata / self.vscale.vfactor10,
 421                                     norm=norm,
 422                                     extent=extent_val,
 423                                     **kwargs)
 424        else:
 425            self.im = self.ax.imshow(valdata.value / self.vscale.vfactor10,
 426                                     norm=norm,
 427                                     extent=extent_val,
 428                                     **kwargs)
 429
 430        # Plot self contours
 431        if self_contours:
 432            self.plot_contours(valdata,
 433                               wcs=valwcs,
 434                               extent=extent,
 435                               rms=rms,
 436                               levels=contour_levels,
 437                               colors=contour_colors,
 438                               nsigma=contour_nsigma,
 439                               negative_nsigma=contour_negative_nsigma,
 440                               nsigmalevel=contour_nsigmalevel,
 441                               linewidths=contour_linewidths,
 442                               stretch=contour_stretch,
 443                               zorder=zorder+1)

 445    def plot_contours(self,
 446                      data: Map,
 447                      wcs: Optional[apy_wcs.WCS] = None,
 448                      extent: Optional[Tuple[u.Quantity]] = None,
 449                      use_extent: bool = False,
 450                      rms: Optional[u.Quantity] = None,
 451                      levels: Optional[List[u.Quantity]] = None,
 452                      colors: Optional[Sequence[str]] = None,
 453                      nsigma: float = 5.,
 454                      negative_nsigma: Optional[float] = None,
 455                      nsigmalevel: Optional[float] = None,
 456                      stretch: Optional[str] = None,
 457                      ignore_units: bool = False,
 458                      **kwargs):
 459        """Plot a contour map.
 460
 461        Args:
 462          data: input map.
 463          wcs: optional; WCS object of the map.
 464          extent: optional; `xy` range of the data (left, right, bottom, top).
 465          use_extent: optional; determine extent from data and ignore wcs.
 466          rms: optional; map noise level.
 467          levels: optional; contour levels.
 468          colors: optional; contour colors.
 469          nsigma: optional; level of the lowest contour over rms.
 470          negative_nsigma: optional; level of the highest negative contour.
 471          nsigmalevel: optional; plot only one contour at this level times the
 472            rms value.
 473          stretch: optional; stretch of contour levels (default:
 474            `axes_props.stretch`)
 475          ignore_units: optional; ignore data units?
 476          **kwargs: optional; additional arguments for `pyplot.contours`.
 477        """
 478        # Validate data
 479        valdata, valwcs = self._validate_data(data, wcs,
 480                                              ignore_units=ignore_units)
 481
 482        # Stretch
 483        if stretch is None:
 484            stretch = self.stretch
 485
 486        # Check extent
 487        if extent is not None:
 488            extent_val = self._validate_extent(extent)
 489            self._log.info('Contour validated extent: %s', extent_val)
 490        elif extent is None and use_extent:
 491            extent = get_extent(data, wcs=wcs)
 492            self._log.info('Contour extent from data: %s', extent)
 493            extent_val = self._validate_extent(extent)
 494            self._log.info('Contour validated extent: %s', extent_val)
 495        else:
 496            extent_val = None
 497
 498        # Levels
 499        if levels is None:
 500            try:
 501                if nsigmalevel is not None:
 502                    nlevels = 1
 503                else:
 504                    nlevels = None
 505                levels = auto_levels(data=valdata,
 506                                     rms=rms,
 507                                     nsigma=nsigma,
 508                                     nsigmalevel=nsigmalevel,
 509                                     nlevels=nlevels,
 510                                     stretch=stretch,
 511                                     negative_nsigma=negative_nsigma,
 512                                     log=self._log.info)
 513                levels_val = levels.to(valdata.unit).value
 514            except ValueError:
 515                return None
 516        else:
 517            levels_val = levels.to(valdata.unit).value
 518
 519        # Color map
 520        if 'cmap' not in kwargs:
 521            kwargs['colors'] = colors or self.skeleton.get('data',
 522                                                           'contour_colors')
 523        elif 'norm' not in kwargs:
 524            kwargs['norm'] = self.vscale.get_normalization()
 525        else:
 526            pass
 527
 528        # Plot
 529        kwargs.setdefault('zorder', 0)
 530        if valwcs is not None and extent_val is None:
 531            return super().contour(valdata,
 532                                   is_image=True,
 533                                   levels=levels_val,
 534                                   transform=self.get_transform(valwcs),
 535                                   **kwargs)
 536        else:
 537            return super().contour(valdata,
 538                                   is_image=True,
 539                                   levels=levels_val,
 540                                   extent=extent_val,
 541                                   **kwargs)

 543    def _plot_artist(self, artist: str) -> None:
 544        """Plot each value of the artist"""
 545        for position, props in zip(self.artists[artist]['positions'],
 546                                   self.artists[artist]['properties']):
 547            if self.radesys:
 548                pos = position.transform_to(self.radesys)
 549            else:
 550                try:
 551                    position.ra = position.ra.to(self.axes_props.xunit)
 552                    position.dec = position.dec.to(self.axes_props.yunit)
 553                    pos = position
 554                except AttributeError:
 555                    pos = position
 556            if artist == 'scatters':
 557                self.scatter(pos.ra, pos.dec, **props)
 558            elif artist == 'texts':
 559                text = props.pop('text')
 560                self.text(pos.ra, pos.dec, text, nphys_args=2, **props)
 561            elif artist == 'arrows':
 562                pa = props.pop('pa')
 563                self.arrow(pos.ra, pos.dec, pa, **props)
 564            elif artist == 'scale':
 565                distance = props.pop('distance')
 566                self.scale(pos.ra, pos.dec, distance, **props)
 567            elif artist == 'regions':
 568                self.region(pos.ra, pos.dec, **props)
 569            elif artist == 'hlines':
 570                pos = position.to(self.axes_props.yunit)
 571                self.axhline(pos.value, **props)
 572            elif artist == 'vlines':
 573                pos = position.to(self.axes_props.xunit)
 574                self.axvline(pos.value, **props)
 575            elif artist == 'axlines':
 576                slope = props.pop('slope')
 577                self.axline(pos.ra, pos.dec, slope=slope, **props)

 611        If ticks are not given, they will be determined from the other

 230    def _validate_data(self, data: Map,
 231                       wcs: apy_wcs.WCS,
 232                       ignore_units: bool = False
 233                       ) -> Tuple[u.Quantity, apy_wcs.WCS]:
 234        """Validate input data.
 235
 236        Convert input data to a quantity with the same unit as `self.bunit` or
 237        determines `self.bunit` from header. If the unit of the data cannot be
 238        determined it is assumed to be in the same units as `self.bunit` or is
 239        considered dimensionless.
 240
 241        Args:
 242          data: input data.
 243          wcs: WCS of the data.
 244          ignore_units: optional; ignore data units?
 245        """
 246        if hasattr(data, 'unit'):
 247            # Check bunit
 248            if not ignore_units and self.vscale.unit is None:
 249                self.vscale.set_unit(data.unit)
 250                self._log.info('Setting bunit to data unit: %s', data.unit)
 251            elif not ignore_units:
 252                valdata = data.to(self.vscale.unit)
 253            else:
 254                valdata = data
 255        elif hasattr(data, 'header'):
 256            # Check bunit
 257            bunit = u.Unit(data.header.get('BUNIT', 1))
 258            valdata = np.squeeze(data.data) * bunit
 259            if not ignore_units and self.vscale.unit is None:
 260                self.vscale.set_unit(bunit)
 261                self._log.info('Setting bunit to header unit: %s', bunit)
 262            elif not ignore_units:
 263                self._log.info('Converting data unit: %s -> %s',
 264                               valdata.unit, self.vscale.unit)
 265                valdata = valdata.to(self.vscale.unit)
 266            else:
 267                pass
 268
 269            # Check wcs
 270            if wcs is None:
 271                wcs = apy_wcs.WCS(data.header, naxis=2)
 272
 273            # Set RADESYS
 274            if self.radesys is None:
 275                try:
 276                    self.radesys = data.header['RADESYS'].lower()
 277                    if self.radesys.upper() == 'J2000':
 278                        self.radesys = 'fk5'
 279                    self._log.info('Setting RADESYS: %s', self.radesys)
 280                except KeyError:
 281                    self._log.warning('Map does not have RADESYS')
 282                    self.radesys = ''
 283        elif not ignore_units:
 284            if self.vscale.unit is None:
 285                self._log.warning('Setting data as dimensionless')
 286                self.vscale.unit = u.Unit(1)
 287            else:
 288                self._log.warning('Assuming data in %s units', self.vscale.unit)
 289            valdata = data * self.vscale.unit
 290        else:
 291            self._log.warning('Setting data as dimensionless')
 292            valdata = data * u.Unit(1)
 293
 294        # Check wcs
 295        if wcs is None:
 296            self._log.warning('WCS is None')
 297
 298        return valdata, wcs
