131        """Configure the plot.
132
133        Configures several aspects of the axes: limits, scales, labels and

136        Args:
137          xticks: optional; x ticks values.
138          yticks: optional; y ticks values.
139          minor_xticks: optional; minor x ticks values.

144        if self.is_config:

149        if axes_props:
150            self._log.info('Replacing plot configs: %s', axes_props)
151            props = dataclasses.replace(self.axes_props, **axes_props)
152        else:
153            props = self.axes_props
154
155        # Limits
156        if props.xlim is not None:
157            self._log.info('Setting xlim: %s', props.xlim)
158            self.ax.set_xlim(**props.xlim)
159        if props.ylim is not None:

164        if props.xscale == 'log':

167        if props.yscale == 'log':
168            self.ax.set_yscale('log')
169            self.ax.yaxis.set_major_formatter(tick_formatter('log'))
170
171        # Labels
172        self.set_axlabels(xlabel=props.xlabel, ylabel=props.ylabel)
173
174        # Ticks
175        if xticks:
176            self.ax.set_xticks(xticks)
177            fmt = props.xticks_fmt
178            self.ax.set_xticklabels([fmt.format(x) for x in xticks])
179        elif not props.set_xticks:
180            self.ax.set_xticklabels([''] * len(self.ax.get_xticks()))
181        if minor_xticks:
182            self.ax.set_xticks(minor_xticks, minor=True)
183        if yticks:

472    def arrow(self,
473              arrow: Union[str, Tuple[float]],
474              arrowprops: Optional[dict] = None,
475              color: Optional[str] = None,
476              **kwargs) -> None:
477        """Draw an arrow.
478
479        An arrow can be specified by the following combiantions:
480
481          - (PA,)
482          - (x, y, PA)
483          - (PA, length)
484          - (x, y, PA, length)
485
486        with PA the position angle (from axis y>0 towards axis x<0). The
487        positions (x, y) are in axes fraction and their default value is
488        (0.5, 0.5). Similarly, the default legth is 0.5, i.e. half the axis
489        length.
490
491        Args:
492          arrow: arrow position, angle and/or legth.
493          arrowprops: optional; arrow properties.
494          color: optional; use the default arrowprops but replace the color.
495          kwargs: other arrow properties. See `matplotlib.pyplot.annotate`.
496        """
497        # Check input
498        if arrowprops is None:
499            arrowprops = {'arrowstyle':'->', 'fc':'k', 'ec':'k', 'ls':'-',
500                          'lw':2}
501
502        # Arrow specified as PA or (x y PA) or (PA len) or (x y PA len)
503        try:
504            arrow = tuple(map(float, arrow.split()))
505        except AttributeError:
506            pass
507        x0, y0, l = 0.5, 0.5, 0.5
508        if len(arrow) == 1:
509            pa = arrow[0]
510        elif len(arrow) == 2:
511            pa, l = arrow
512        elif len(arrow) == 3:
513            x0, y0, pa = arrow
514        elif len(arrow) == 4:
515            x0, y0, pa, l = arrow
516        else:
517            raise ValueError(f'Cannot configure arrow: {arrow}')
518        l = l * 0.5
519        pa = pa + 90.
520
521        # Locations
522        dx = l * np.cos(np.radians(pa))
523        dy = l * np.sin(np.radians(pa))
524        xy = (x0 + dx, y0 + dy)
525        xytext = (x0 - dx, y0 - dy)
526
527        # Check colors
528        if color and color != arrowprops['fc']:
529            arrowprops['fc'] = color
530            arrowprops['ec'] = color
531        elif color is None:
532            color = arrowprops['fc']
533
534        # Draw
535        self.annotate('', xy=xy, xytext=xytext, xycoords='axes fraction',
536                      arrowprops=arrowprops, color=color, **kwargs)

587    def _simple_plt(self,
588                    fn: Callable[[], Plot],
589                    *args,
590                    nphys_args: Optional[int] = None,
591                    is_image: bool = False,
592                    **kwargs) -> Plot:
593        """Plot and assign label.
594
595        It verifies that args are in the correct units of the corresponding
596        axis. The parameter `nphys_args` can be used to specify the number of
597        `args` that are physical quantities. If `args` consist of an image,
598        then it is assumed that it is in the correct units and the `is_image`
599        keyword can be used to skip the unit checking.
600
601        Args:
602          fn: plotting function.
603          args: arguments for the function.
604          nphys_args: optional; number of physical quantities.
605          is_image: optional; is the input args a 2-D image?
606          kwargs: keyword arguments for the function.
607        """
608        # Verify nphys_args
609        if nphys_args is None:
610            nphys_args = len(args)
611
612        # Cases
613        xunit = self.axes_props.xunit
614        yunit = self.axes_props.yunit
615        if is_image:
616            phys_args = (args[0].value,)
617        elif nphys_args == 1:
618            phys_args = (args[0].to(yunit).value,)
619        elif nphys_args == 2:
620            phys_args = (args[0].to(xunit).value,
621                         args[1].to(yunit).value)
622        elif nphys_args == 3:
623            phys_args = (args[0].to(xunit).value,
624                         args[1].to(yunit).value,
625                         args[2].to(yunit).value)
626        else:
627            raise ValueError('Could not convert values')
628
629        # Non physical arguments
630        non_phys_args = args[nphys_args:]
631        fn_args = phys_args + non_phys_args
632
633        return self.insert_plt(kwargs.get('label', None),
634                               fn(*fn_args, **kwargs))

478        """Draw an arrow.
479
480        An arrow can be specified by the following combiantions:
481
482          - (PA,)
483          - (x, y, PA)
484          - (PA, length)
485          - (x, y, PA, length)
486

506        except AttributeError:
507            pass
508        x0, y0, l = 0.5, 0.5, 0.5
509        if len(arrow) == 1:
510            pa = arrow[0]
511        elif len(arrow) == 2:
512            pa, l = arrow
513        elif len(arrow) == 3:
514            x0, y0, pa = arrow
515        elif len(arrow) == 4:
516            x0, y0, pa, l = arrow
517        else:

521

531            arrowprops['ec'] = color
532        elif color is None:
533            color = arrowprops['fc']
